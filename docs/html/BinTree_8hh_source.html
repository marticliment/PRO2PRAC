307template <class T>
308void BinTree<T>::readBinTreePostOrder(std::istream &is, BinTree<T> &t)
309{
310 int size;
311 std::stack<BinTree<T> > s;
312
313 is >> size;
314 while (size > 0) {
315 T node;
316 int numchildren; // 0: zero, -1: only left, 1: only right, 2: two
317 is >> node >> numchildren;
318 if (numchildren == 0) {
319 s.push(BinTree<T>(node, BinTree<T>(), BinTree<T>()));
320 } else if (numchildren == -1) {
321 BinTree<T> left = s.top();
322 s.pop();
323 s.push(BinTree<T>(node, left, BinTree<T>()));
324 } else if (numchildren == 1) {
325 BinTree<T> right = s.top();
326 s.pop();
327 s.push(BinTree<T>(node, BinTree<T>(), right));
328 } else {
329 BinTree<T> right = s.top();
330 s.pop();
331 BinTree<T> left = s.top();
332 s.pop();
333 s.push(BinTree<T>(node, left, right));
334 }
335 --size;
336 }
337 if (not s.empty()) t = s.top();
338}

566 if (t.empty()) return vector<string> ();

568 int rootrelativepos = (int(rootstring.size())-1)/2;
569 vector<string> leftresult = generateVisualFormatRec(t.left());

576 rightrootpos = obtainRootPositionInString(rightresult[0]);
577 vector<string> result;
578 if (int(leftresult.size()) == 0 and int(rightresult.size()) == 0) {
579 result.push_back(rootstring);
580 return result;
581 }
582 if (int(leftresult.size()) == 0) {
583 int rootpos = rightrootpos-MINNUMHYPENSDESCENDANT-1;
584 if (rootpos-rootrelativepos < 0) {
585 insertSpaceVectorString(-(rootpos-rootrelativepos), rightresult);
586 rightrootpos += -(rootpos-rootrelativepos);
587 rootpos = rootrelativepos;
588 }
589 result.push_back(string(rootpos-rootrelativepos, ' ') + rootstring);
590 result.push_back(string(rootpos, ' ') + string(1, '|'));
591 result.push_back(string(rootpos+1, ' ') + string(rightrootpos-rootpos-1, '-'));
592 result.push_back(string(rightrootpos, ' ') + string(1, '|'));
593 for (int i = 0; i < int(rightresult.size()); ++i)
594 result.push_back(rightresult[i]);
595 return result;
596 }
597 if (int(rightresult.size()) == 0) {
598 int rootpos = leftrootpos+MINNUMHYPENSDESCENDANT+1;
599 if (rootpos-rootrelativepos < 0) {
600 insertSpaceVectorString(-(rootpos-rootrelativepos), leftresult);
601 leftrootpos += -(rootpos-rootrelativepos);
602 rootpos = rootrelativepos;
603 }
604 result.push_back(string(rootpos-rootrelativepos, ' ') + rootstring);

606 result.push_back(string(leftrootpos+1, ' ') + string(max(0,rootpos-leftrootpos-1), '-'));
607 result.push_back(string(leftrootpos, ' ') + string(1, '|'));
608 for (int i = 0; i < int(leftresult.size()); ++i)
609 result.push_back(leftresult[i]);
610 return result;
611 }

613 for (int i = 0; i < int(leftresult.size()) and i < int(rightresult.size()); ++i)
614 addspace = max(addspace, int(leftresult[i].size())-firstNonBlankPosInString(rightresult[i])+MINSPACEBETWEENSUBTREES);
615 if ((rightrootpos+addspace-leftrootpos)%2 == 1) addspace++;

618 rightrootpos += addspace;
619 } else if (addspace < 0) {
620 insertSpaceVectorString(-addspace, leftresult);
621 leftrootpos += -addspace;
622 }
623 int rootpos = (leftrootpos+rightrootpos)/2;
624 if (rootpos-rootrelativepos < 0) {
625 insertSpaceVectorString(-(rootpos-rootrelativepos), leftresult);
626 insertSpaceVectorString(-(rootpos-rootrelativepos), rightresult);
627 leftrootpos += -(rootpos-rootrelativepos);
628 rightrootpos += -(rootpos-rootrelativepos);
629 rootpos = rootrelativepos;
630 }
631 result.push_back(string(rootpos-rootrelativepos, ' ') + rootstring);
632 result.push_back(string(rootpos, ' ') + string(1, '|'));
633 result.push_back(string(leftrootpos+1, ' ') + string(rootpos-leftrootpos-1, '-') + string(1, ' ') +
634 string(max(0,rightrootpos-rootpos-1), '-'));
635 result.push_back(string(leftrootpos, ' ') + string(1, '|') + string(rightrootpos-leftrootpos-1, ' ') + string(1, '|'));
636 //cout << "Adding left and right result" << endl;

638 if (i < int(leftresult.size()) and i < int(rightresult.size())) {
